Ejercicio 4: Factorial de un n칰mero

Problema: Calcular el factorial de un n칰mero entero positivo usando un bucle for.
游눹 factorial.cpp 
/*
Descripci칩n: Calcular el factorial de un 
n칰mero entero positivo usando bucle for.
*/

#include
using namespace std;

int main(){
    
    //DEFINIR VARIABLES
    int numero;
    long long factorial = 1;
    
    //ENTRADA DE DATOS
    cout << "INGRESE UN NUMERO POSITIVO: ";
    cin >> numero;
    
    //VALIDACI칍N
    if(numero < 0){
        cout << "ERROR: El factorial no existe para numeros negativos" << endl;
        return 1;
    }
    
    //PROCESO
    for(int i = 1; i <= numero; i++){
        factorial *= i;
    }
    
    //SALIDA
    cout << "EL FACTORIAL DE " << numero << " ES: " << factorial << endl;
    
    return 0;
}

Descripci칩n: Mostrar la tabla de multiplicar 
de un n칰mero del 1 al 10.
*/

#include
using namespace std;

int main(){
    
    //DEFINIR VARIABLES
    int numero;
    
    //ENTRADA DE DATOS
    cout << "INGRESE UN NUMERO PARA SU TABLA: ";
    cin >> numero;
    
    //PROCESO Y SALIDA
    cout << "\n--- TABLA DEL " << numero << " ---" << endl;

        cout << numero << " x " << i << " = " << (numero * i) << endl;
    }
    
    return 0;
}
