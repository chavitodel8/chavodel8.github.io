Descripci칩n: Calcular el factorial de un

*/

#include
using namespace std;

int main(){
    
    //DEFINIR VARIABLES
    int numero;
    long long factorial = 1;
    
    //ENTRADA DE DATOS
    cout << "INGRESE UN NUMERO POSITIVO: ";
    cin >> numero;
    
    //VALIDACI칍N
    if(numero < 0){
        cout << "ERROR: El factorial no existe para numeros negativos" << endl;
        return 1;
    }
    
    //PROCESO
    for(int i = 1; i <= numero; i++){
        factorial *= i;
    }
    
    //SALIDA
    cout << "EL FACTORIAL DE " << numero << " ES: " << factorial << endl;
    
    return 0;
}

Descripci칩n: Generar los primeros N 
t칠rminos de la serie Fibonacci.
*/

#include
using namespace std;

int main(){
    
    //DEFINIR VARIABLES
    int n, primero = 0, segundo = 1, siguiente;
    
    //ENTRADA DE DATOS
    cout << "INGRESE CUANTOS TERMINOS DESEA: ";
    cin >> n;
    
    //VALIDACI칍N
    if(n <= 0){
        cout << "ERROR: Ingrese un numero positivo" << endl;
        return 1;
    }
    
    //SALIDA
    cout << "\n--- SERIE FIBONACCI ---" << endl;
    
    for(int i = 1; i <= n; i++){
        if(i == 1){
            cout << primero << " ";
        } else if(i == 2){
            cout << segundo << " ";
        } else {
            siguiente = primero + segundo;
            cout << siguiente << " ";
            primero = segundo;
            segundo = siguiente;
        }
    }
    
    cout << endl;
    return 0;
}
